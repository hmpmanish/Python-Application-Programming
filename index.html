def fib(n):
    if n <= 0: return 0
    elif n == 1: return 1
    else: return fib(n-1) + fib(n-2)

n = int(input("Number of terms: "))
sequence = [fib(i) for i in range(n)]
print("Fibonacci:", sequence)

# Optimized with memoization
from functools import lru_cache

@lru_cache
def fib_fast(n):
    if n <= 1: return n
    return fib_fast(n-1) + fib_fast(n-2)

Operator Overloading for Point Class
Python 
import math

class Point:
    def __init__(self, x, y):
        self.x = x
        self.y = y

    def __str__(self):
        return f"Point({self.x}, {self.y})"

    # a) Overload (-) for distance
    def __sub__(self, other):
        return math.sqrt((self.x - other.x)**2 + (self.y - other.y)**2)

    def dist_origin(self):
        return math.sqrt(self.x**2 + self.y**2)

    # b) Overload (>=) for distance from origin
    def __ge__(self, other):
        return self.dist_origin() >= other.dist_origin()

P1, P2, P3 = Point(3,4), Point(6,8), Point(5,12)
print(f"P1-P2 distance: {P1-P2:.2f}")
print(f"P3 >= P1: {P3>=P1}")
Sample Output
P1-P2 distance: 5.00
P3 >= P1: True

class InvalidMarksError(Exception):
    def __init__(self, subject, marks):
        self.message = f"Invalid {subject}: {marks}. Must be 0-100."
        super().__init__(self.message)

class Student:
    def __init__(self, name, roll, phy, chem, math):
        self.name = name
        self.roll = roll
        for subj, marks in [("Physics",phy),("Chemistry",chem),("Math",math)]:
            if marks < 0 or marks > 100:
                raise InvalidMarksError(subj, marks)
        self.phy, self.chem, self.math = phy, chem, math

    def average(self):
        return (self.phy + self.chem + self.math) / 3

try:
    s = Student("Rahul", "001", 85, 92, 78)
    print(f"Average: {s.average():.2f}%")
except InvalidMarksError as e:
    print(f"Error: {e}")
Sample Output
Average: 85.00%

Bank Account Management (Abstract Class, Polymorphism)
Python 
from abc import ABC, abstractmethod

class Account(ABC):
    def __init__(self, acc_no, holder, balance):
        self.__acc_no = acc_no  # Private
        self.__holder = holder
        self.__balance = balance

    @abstractmethod
    def calculate_interest(self): pass

    def get_balance(self): return self.__balance

class SavingsAccount(Account):
    def calculate_interest(self):
        interest = self.get_balance() * 0.04
        print(f"Savings Interest (4%): Rs.{interest:.2f}")
        return interest

class CurrentAccount(Account):
    def calculate_interest(self):
        interest = self.get_balance() * 0.02
        print(f"Current Interest (2%): Rs.{interest:.2f}")
        return interest

# Runtime polymorphism
accounts = [SavingsAccount("SA001","Alice",50000),
            CurrentAccount("CA001","Bob",100000)]
for acc in accounts:
    acc.calculate_interest()  # Same call, different behavior
Sample Output
Savings Interest (4%): Rs.2000.00
Current Interest (2%): Rs.2000.00
